import { useState, useEffect } from 'react'
import { useAuth } from '../hooks/useAuth'
import { useNavigate, useParams } from 'react-router-dom'
import { apiService, type ProjectRequest } from '../services/api'

interface FormData {
  name: string
  description: string
  shortDescription: string
  price: number
  currency: string
  originalPrice: number
  mainImage: string
  screenshots: string[]
  videoUrl: string
  category: string
  tags: string[]
  targetAudience: string
  demoUrl: string
  documentationUrl: string
  sourceCodeUrl: string
  supportEmail: string
  keywords: string[]
}

const CATEGORIES = [
  'WEB_DEVELOPMENT',
  'MOBILE_APPS',
  'DESKTOP_SOFTWARE',
  'DESIGN_TEMPLATES',
  'PRODUCTIVITY_TOOLS',
  'BUSINESS_TOOLS',
  'EDUCATION',
  'ECOMMERCE',
  'ENTERTAINMENT',
  'GAMES',
  'OTHER'
]

const CATEGORY_LABELS: Record<string, string> = {
  'WEB_DEVELOPMENT': 'Desarrollo Web',
  'MOBILE_APPS': 'Aplicaciones M√≥viles',
  'DESKTOP_SOFTWARE': 'Software de Escritorio',
  'DESIGN_TEMPLATES': 'Plantillas de Dise√±o',
  'PRODUCTIVITY_TOOLS': 'Herramientas de Productividad',
  'BUSINESS_TOOLS': 'Herramientas de Negocio',
  'EDUCATION': 'Educaci√≥n',
  'ECOMMERCE': 'E-commerce',
  'ENTERTAINMENT': 'Entretenimiento',
  'GAMES': 'Juegos',
  'OTHER': 'Otros'
}

interface ProjectFormProps {
  mode: 'create' | 'edit'
}

const ProjectForm = ({ mode }: ProjectFormProps) => {
  const { user, isAuthenticated } = useAuth()
  const navigate = useNavigate()
  const { id } = useParams<{ id: string }>()
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [isLoading, setIsLoading] = useState(mode === 'edit')
  const [isValidatingGitHub, setIsValidatingGitHub] = useState(false)
  const [currentTag, setCurrentTag] = useState('')
  const [currentKeyword, setCurrentKeyword] = useState('')
  const [formData, setFormData] = useState<FormData>({
    name: '',
    description: '',
    shortDescription: '',
    price: 0,
    currency: 'USD',
    originalPrice: 0,
    mainImage: '',
    screenshots: [],
    videoUrl: '',
    category: '',
    tags: [],
    targetAudience: '',
    demoUrl: '',
    documentationUrl: '',
    sourceCodeUrl: '',
    supportEmail: user?.email || '',
    keywords: []
  })
  const [errors, setErrors] = useState<Record<string, string>>({})

  // Cargar datos si es edici√≥n
  useEffect(() => {
    if (mode === 'edit' && id && isAuthenticated) {
      const loadProject = async () => {
        try {
          const project = await apiService.getProjectById(id)
          if (!project) {
            navigate('/developer-dashboard')
            return
          }
          if (project.developerId !== user?.id) {
            navigate('/developer-dashboard')
            return
          }
          setFormData({
            name: project.name || '',
            description: project.description || '',
            shortDescription: project.shortDescription || '',
            price: project.price || 0,
            currency: project.currency || 'USD',
            originalPrice: project.originalPrice || project.price || 0,
            mainImage: project.mainImage || '',
            screenshots: project.screenshots || [],
            videoUrl: project.videoUrl || '',
            category: project.category || '',
            tags: project.tags || [],
            targetAudience: project.targetAudience || '',
            demoUrl: project.demoUrl || '',
            documentationUrl: project.documentationUrl || '',
            sourceCodeUrl: project.sourceCodeUrl || '',
            supportEmail: project.supportEmail || user?.email || '',
            keywords: project.keywords || []
          })
        } catch {
          navigate('/developer-dashboard')
        } finally {
          setIsLoading(false)
        }
      }
      loadProject()
    }
  }, [mode, id, isAuthenticated, user, navigate])

  const validateGitHubRepository = async (url: string): Promise<boolean> => {
    if (!url) return false
    
    const githubRegex = /^https:\/\/github\.com\/([a-zA-Z0-9_-]+)\/([a-zA-Z0-9_-]+)$/
    const match = githubRegex.exec(url)
    
    if (!match) return false
    
    const [, owner, repo] = match
    
    try {
      const response = await fetch(`https://api.github.com/repos/${owner}/${repo}`, {
        headers: {
          'Accept': 'application/vnd.github.v3+json'
        }
      })
      return response.status === 200
    } catch (error) {
      console.error('Error validando repositorio de GitHub:', error)
      return false
    }
  }

  // Verificar que el usuario est√° autenticado y es desarrollador
  const isDeveloper = user?.type === 'developer'

  if (isLoading) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600 mx-auto mb-4"></div>
          <p className="text-gray-600">Cargando proyecto...</p>
        </div>
      </div>
    )
  }

  if (!isAuthenticated) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="text-center">
          <span className="text-6xl mb-4 block">üîí</span>
          <h2 className="text-2xl font-bold text-gray-800 mb-4">Acceso Requerido</h2>
          <p className="text-gray-600 mb-4">Necesitas iniciar sesi√≥n para {mode === 'edit' ? 'editar' : 'publicar'} proyectos.</p>
          <button
            onClick={() => navigate('/login')}
            className="bg-blue-600 hover:bg-blue-700 text-white px-6 py-2 rounded-lg transition-colors duration-200"
          >
            Iniciar Sesi√≥n
          </button>
        </div>
      </div>
    )
  }

  if (!isDeveloper) {
    return (
      <div className="min-h-screen bg-gray-50 flex items-center justify-center">
        <div className="text-center">
          <span className="text-6xl mb-4 block">üë®‚Äçüíª</span>
          <h2 className="text-2xl font-bold text-gray-800 mb-4">Solo para Desarrolladores</h2>
          <p className="text-gray-600 mb-4">Necesitas ser un desarrollador verificado para {mode === 'edit' ? 'editar' : 'publicar'} proyectos.</p>
          <button 
            onClick={() => navigate('/developer-dashboard')}
            className="bg-green-600 hover:bg-green-700 text-white px-6 py-2 rounded-lg transition-colors duration-200"
          >
            Ir al Dashboard
          </button>
        </div>
      </div>
    )
  }

  const addTag = () => {
    if (currentTag.trim() && !formData.tags.includes(currentTag.trim().toLowerCase())) {
      setFormData(prev => ({
        ...prev,
        tags: [...prev.tags, currentTag.trim().toLowerCase()]
      }))
      setCurrentTag('')
    }
  }

  const removeTag = (index: number) => {
    setFormData(prev => ({
      ...prev,
      tags: prev.tags.filter((_, i) => i !== index)
    }))
  }

  const addKeyword = () => {
    if (currentKeyword.trim() && !formData.keywords.includes(currentKeyword.trim().toLowerCase())) {
      setFormData(prev => ({
        ...prev,
        keywords: [...prev.keywords, currentKeyword.trim().toLowerCase()]
      }))
      setCurrentKeyword('')
    }
  }

  const removeKeyword = (index: number) => {
    setFormData(prev => ({
      ...prev,
      keywords: prev.keywords.filter((_, i) => i !== index)
    }))
  }

  const validateForm = async (): Promise<boolean> => {
    const newErrors: Record<string, string> = {}

    // Validaciones b√°sicas
    if (!formData.name.trim()) newErrors.name = 'El nombre es requerido'
    if (formData.name.trim().length < 3) newErrors.name = 'El nombre debe tener al menos 3 caracteres'
    if (formData.name.trim().length > 100) newErrors.name = 'El nombre no puede exceder 100 caracteres'
    
    if (!formData.description.trim()) newErrors.description = 'La descripci√≥n es requerida'
    if (formData.description.trim().length < 50) newErrors.description = 'La descripci√≥n debe tener al menos 50 caracteres'
    if (formData.description.trim().length > 2000) newErrors.description = 'La descripci√≥n no puede exceder 2000 caracteres'
    
    if (!formData.shortDescription.trim()) newErrors.shortDescription = 'La descripci√≥n corta es requerida'
    if (formData.shortDescription.trim().length < 20) newErrors.shortDescription = 'La descripci√≥n corta debe tener al menos 20 caracteres'
    if (formData.shortDescription.trim().length > 200) newErrors.shortDescription = 'La descripci√≥n corta no puede exceder 200 caracteres'
    
    if (formData.price < 0) newErrors.price = 'El precio debe ser mayor o igual a 0'
    if (formData.price > 10000) newErrors.price = 'El precio no puede exceder $10,000'
    
    if (!formData.category) newErrors.category = 'La categor√≠a es requerida'
    if (!CATEGORIES.includes(formData.category)) newErrors.category = 'Categor√≠a inv√°lida'
    
    if (!formData.targetAudience.trim()) newErrors.targetAudience = 'La audiencia objetivo es requerida'
    if (formData.targetAudience.trim().length < 10) newErrors.targetAudience = 'La audiencia objetivo debe ser m√°s descriptiva (m√≠nimo 10 caracteres)'
    
    if (!formData.sourceCodeUrl.trim()) newErrors.sourceCodeUrl = 'La URL del c√≥digo fuente es requerida'
    if (!formData.supportEmail.trim()) newErrors.supportEmail = 'El email de soporte es requerido'
    if (!formData.mainImage.trim()) newErrors.mainImage = 'La imagen principal es requerida'
    
    // Validaciones espec√≠ficas para tags y keywords
    if (formData.tags.length === 0) newErrors.tags = 'Agrega al menos un tag'
    if (formData.tags.length < 2) newErrors.tags = 'Se requieren al menos 2 tags para mejor categorizaci√≥n'
    if (formData.tags.length > 10) newErrors.tags = 'No puedes agregar m√°s de 10 tags'
    
    if (formData.keywords.length === 0) newErrors.keywords = 'Agrega al menos una palabra clave'
    if (formData.keywords.length < 3) newErrors.keywords = 'Se requieren al menos 3 palabras clave para mejor b√∫squeda'
    if (formData.keywords.length > 15) newErrors.keywords = 'No puedes agregar m√°s de 15 palabras clave'

    // Validar email
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
    if (formData.supportEmail && !emailRegex.test(formData.supportEmail)) {
      newErrors.supportEmail = 'Email de soporte inv√°lido'
    }

    // Validar URLs m√°s estrictamente
    const urlRegex = /^https?:\/\/.+/
    const githubRegex = /^https:\/\/github\.com\/[a-zA-Z0-9_-]+\/[a-zA-Z0-9_-]+$/
    
    if (formData.sourceCodeUrl && !urlRegex.test(formData.sourceCodeUrl)) {
      newErrors.sourceCodeUrl = 'URL del c√≥digo fuente inv√°lida (debe incluir http:// o https://)'
    } else if (formData.sourceCodeUrl && !githubRegex.test(formData.sourceCodeUrl)) {
      newErrors.sourceCodeUrl = 'La URL debe ser un repositorio de GitHub v√°lido (ejemplo: https://github.com/usuario/proyecto)'
    } else if (formData.sourceCodeUrl) {
      // Validar que el repositorio existe en GitHub
      setIsValidatingGitHub(true)
      const repoExists = await validateGitHubRepository(formData.sourceCodeUrl)
      setIsValidatingGitHub(false)
      
      if (!repoExists) {
        newErrors.sourceCodeUrl = 'El repositorio de GitHub no existe o no es p√∫blico'
      }
    }
    
    if (formData.mainImage && !urlRegex.test(formData.mainImage)) {
      newErrors.mainImage = 'URL de imagen principal inv√°lida (debe incluir http:// o https://)'
    }
    
    // Hacer demoUrl obligatorio
    if (!formData.demoUrl?.trim()) {
      newErrors.demoUrl = 'La URL de la aplicaci√≥n es obligatoria'
    } else if (!urlRegex.test(formData.demoUrl)) {
      newErrors.demoUrl = 'URL de aplicaci√≥n inv√°lida (debe incluir http:// o https://)'
    }
    
    if (formData.documentationUrl?.trim() && !urlRegex.test(formData.documentationUrl)) {
      newErrors.documentationUrl = 'URL de documentaci√≥n inv√°lida (debe incluir http:// o https://)'
    }
    if (formData.videoUrl?.trim() && !urlRegex.test(formData.videoUrl)) {
      newErrors.videoUrl = 'URL de video inv√°lida (debe incluir http:// o https://)'
    }

    // Validar screenshots
    if (formData.screenshots.length > 0) {
      const invalidScreenshots = formData.screenshots.filter(url => url.trim() && !urlRegex.test(url.trim()))
      if (invalidScreenshots.length > 0) {
        newErrors.screenshots = 'Todas las URLs de screenshots deben incluir http:// o https://'
      }
    }

    setErrors(newErrors)
    return Object.keys(newErrors).length === 0
  }

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    
    if (!isAuthenticated || !user || user.type !== 'developer') {
      setErrors({ submit: 'Debes iniciar sesi√≥n como desarrollador.' })
      return
    }
    
    const isValid = await validateForm()
    if (!isValid) {
      return
    }

    setIsSubmitting(true)

    try {
      const projectData: ProjectRequest = {
        name: formData.name.trim(),
        description: formData.description.trim(),
        shortDescription: formData.shortDescription.trim(),
        price: formData.price,
        currency: formData.currency,
        originalPrice: formData.originalPrice > 0 ? formData.originalPrice : formData.price,
        mainImage: formData.mainImage,
        screenshots: formData.screenshots.filter(url => url.trim() !== ''),
        videoUrl: formData.videoUrl.trim() || undefined,
        category: formData.category,
        tags: formData.tags,
        targetAudience: formData.targetAudience.trim(),
        demoUrl: formData.demoUrl.trim() || undefined,
        documentationUrl: formData.documentationUrl.trim() || undefined,
        sourceCodeUrl: formData.sourceCodeUrl.trim(),
        customCommissionRate: 0.10,
        supportEmail: formData.supportEmail.trim(),
        keywords: formData.keywords
      }

      if (mode === 'edit' && id) {
        await apiService.updateProject(id, projectData)
      } else {
        await apiService.createProject(projectData)
      }
      
      navigate('/developer-dashboard', { 
        state: { message: mode === 'edit' ? '¬°Proyecto actualizado exitosamente!' : '¬°Proyecto publicado exitosamente!' }
      })
      
    } catch (error) {
      console.error('Error al procesar el proyecto:', error)
      if (error instanceof Error) {
        let userFriendlyMessage = error.message
        if (error.message.includes('Errores de validaci√≥n espec√≠ficos')) {
          userFriendlyMessage = error.message.replace('Errores de validaci√≥n espec√≠ficos:', 'Problemas encontrados:')
        } else if (error.message.includes('Errores de validaci√≥n')) {
          userFriendlyMessage = 'Hay problemas con los datos enviados. Revisa los campos marcados arriba.'
        }
        
        setErrors({ submit: userFriendlyMessage })
      } else {
        setErrors({ submit: `Error al ${mode === 'edit' ? 'actualizar' : 'publicar'} el proyecto. Por favor, intenta de nuevo.` })
      }
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <div className="min-h-screen bg-gray-50 pt-24 py-8">
      <div className="container mx-auto px-4 max-w-4xl">
        {/* Header */}
        <div className="mb-8">
          <button
            onClick={() => navigate('/developer-dashboard')}
            className="flex items-center text-blue-600 hover:text-blue-700 mb-4 transition-colors duration-200"
          >
            <svg className="w-5 h-5 mr-2" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M15 19l-7-7 7-7" />
            </svg>
            Volver al Dashboard
          </button>
          
          <h1 className="text-3xl font-bold text-gray-900 mb-2">
            {mode === 'edit' ? 'Editar Proyecto ‚úèÔ∏è' : 'Publicar Nuevo Proyecto üöÄ'}
          </h1>
          <p className="text-gray-600">
            {mode === 'edit' 
              ? 'Actualiza la informaci√≥n de tu proyecto' 
              : 'Completa la informaci√≥n de tu proyecto para publicarlo en la tienda'
            }
          </p>
        </div>

        <form onSubmit={handleSubmit} className="space-y-8">
          {/* Informaci√≥n B√°sica */}
          <div className="bg-white rounded-lg shadow-sm p-6">
            <h2 className="text-xl font-semibold text-gray-900 mb-6">üìã Informaci√≥n B√°sica</h2>
            
            <div className="grid md:grid-cols-2 gap-6">
              <div>
                <label htmlFor="project-name" className="block text-sm font-medium text-gray-700 mb-2">
                  Nombre del Proyecto *
                </label>
                <input
                  id="project-name"
                  type="text"
                  value={formData.name}
                  onChange={(e) => setFormData(prev => ({ ...prev, name: e.target.value }))}
                  className={`w-full px-3 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 ${
                    errors.name ? 'border-red-500' : 'border-gray-300'
                  }`}
                  placeholder="Ej: E-commerce Dashboard"
                />
                {errors.name && <p className="text-red-500 text-sm mt-1">{errors.name}</p>}
              </div>

              <div>
                <label htmlFor="project-category" className="block text-sm font-medium text-gray-700 mb-2">
                  Categor√≠a *
                </label>
                <select
                  id="project-category"
                  value={formData.category}
                  onChange={(e) => setFormData(prev => ({ ...prev, category: e.target.value }))}
                  className={`w-full px-3 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 ${
                    errors.category ? 'border-red-500' : 'border-gray-300'
                  }`}
                >
                  <option value="">Selecciona una categor√≠a</option>
                  {CATEGORIES.map(category => (
                    <option key={category} value={category}>
                      {CATEGORY_LABELS[category]}
                    </option>
                  ))}
                </select>
                {errors.category && <p className="text-red-500 text-sm mt-1">{errors.category}</p>}
              </div>
            </div>

            <div className="mt-6">
              <label htmlFor="short-description" className="block text-sm font-medium text-gray-700 mb-2">
                Descripci√≥n Corta *
              </label>
              <input
                id="short-description"
                type="text"
                value={formData.shortDescription}
                onChange={(e) => setFormData(prev => ({ ...prev, shortDescription: e.target.value }))}
                className={`w-full px-3 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 ${
                  errors.shortDescription ? 'border-red-500' : 'border-gray-300'
                }`}
                placeholder="Ej: Dashboard para e-commerce con React y Node.js"
                maxLength={150}
              />
              {errors.shortDescription && <p className="text-red-500 text-sm mt-1">{errors.shortDescription}</p>}
              <p className="text-xs text-gray-500 mt-1">M√°ximo 150 caracteres</p>
            </div>
          </div>

          {/* Descripci√≥n Completa */}
          <div className="bg-white rounded-lg shadow-sm p-6">
            <h2 className="text-xl font-semibold text-gray-900 mb-6">üìù Descripci√≥n Completa</h2>
            
            <div>
              <label htmlFor="full-description" className="block text-sm font-medium text-gray-700 mb-2">
                Descripci√≥n Detallada *
              </label>
              <textarea
                id="full-description"
                value={formData.description}
                onChange={(e) => setFormData(prev => ({ ...prev, description: e.target.value }))}
                rows={6}
                className={`w-full px-3 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 ${
                  errors.description ? 'border-red-500' : 'border-gray-300'
                }`}
                placeholder="Descripci√≥n detallada del proyecto, sus caracter√≠sticas, beneficios y casos de uso..."
              />
              {errors.description && <p className="text-red-500 text-sm mt-1">{errors.description}</p>}
            </div>
          </div>

          {/* Precio */}
          <div className="bg-white rounded-lg shadow-sm p-6">
            <h2 className="text-xl font-semibold text-gray-900 mb-6">üí∞ Precio</h2>
            
            <div className="grid md:grid-cols-2 gap-6">
              <div>
                <label htmlFor="project-price" className="block text-sm font-medium text-gray-700 mb-2">
                  Precio de Venta (USD) *
                </label>
                <input
                  id="project-price"
                  type="number"
                  min="0"
                  step="0.01"
                  value={formData.price}
                  onChange={(e) => setFormData(prev => ({ ...prev, price: parseFloat(e.target.value) || 0 }))}
                  className={`w-full px-3 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 ${
                    errors.price ? 'border-red-500' : 'border-gray-300'
                  }`}
                  placeholder="79.99"
                />
                {errors.price && <p className="text-red-500 text-sm mt-1">{errors.price}</p>}
              </div>

              <div>
                <label htmlFor="project-original-price" className="block text-sm font-medium text-gray-700 mb-2">
                  Precio Original (Opcional)
                </label>
                <input
                  id="project-original-price"
                  type="number"
                  min="0"
                  step="0.01"
                  value={formData.originalPrice}
                  onChange={(e) => setFormData(prev => ({ ...prev, originalPrice: parseFloat(e.target.value) || 0 }))}
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500"
                  placeholder="99.99"
                />
                <p className="text-xs text-gray-500 mt-1">Para mostrar descuentos</p>
              </div>
            </div>
            
            <div className="mt-4 p-3 bg-blue-50 rounded-lg">
              <p className="text-sm text-blue-700">
                üí° <strong>Comisi√≥n de la plataforma:</strong> 25% sobre cada venta + comisi√≥n de Paypal (variable)
              </p>
            </div>
          </div>

          {/* Audiencia y URLs */}
          <div className="bg-white rounded-lg shadow-sm p-6">
            <h2 className="text-xl font-semibold text-gray-900 mb-6">üéØ Audiencia y Enlaces</h2>
            
            <div className="space-y-6">
              <div>
                <label htmlFor="target-audience" className="block text-sm font-medium text-gray-700 mb-2">
                  Audiencia Objetivo *
                </label>
                <input
                  id="target-audience"
                  type="text"
                  value={formData.targetAudience}
                  onChange={(e) => setFormData(prev => ({ ...prev, targetAudience: e.target.value }))}
                  className={`w-full px-3 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 ${
                    errors.targetAudience ? 'border-red-500' : 'border-gray-300'
                  }`}
                  placeholder="Desarrolladores web, emprendedores digitales, propietarios de tiendas online"
                />
                {errors.targetAudience && <p className="text-red-500 text-sm mt-1">{errors.targetAudience}</p>}
              </div>

              <div className="grid md:grid-cols-2 gap-6">
                <div>
                  <label htmlFor="source-code-url" className="block text-sm font-medium text-gray-700 mb-2">
                    URL del C√≥digo Fuente *
                  </label>
                  <input
                    id="source-code-url"
                    type="url"
                    value={formData.sourceCodeUrl}
                    onChange={(e) => setFormData(prev => ({ ...prev, sourceCodeUrl: e.target.value }))}
                    className={`w-full px-3 py-2 border rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 ${
                      errors.sourceCodeUrl ? 'border-red-500' : 'border-gray-300'
                    }`}
                    placeholder="https://github.com/miusuario/proyecto"
                    disabled={isValidatingGitHub}
                  />
                  {isValidatingGitHub && (
                    <div className="flex items-center mt-1 text-blue-600">
                      <svg className="w-4 h-4 mr-2 animate-spin" fill="none" viewBox="0 0 24 24">
